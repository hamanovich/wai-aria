aria-invalid

Form validation

          <label for="name">Add name</label>
          <input
            name="name"
            id="name"
            aria-required="true"
            aria-invalid="false"
            placeholder="Enter your name"
            onblur="alert('Invalid name entered ❌')"/>

          <label for="email">Add email</label>
          <input
            name="email"
            id="email"
            aria-required="true"
            aria-invalid="true"
            placeholder="Enter e-mail"/>
